Fix for ExpressionChangedAfterItHasBeenCheckedError: Expression has changed after it was checked. when using ngx-bootstrap datepicker

import { Component, OnInit, ChangeDetectorRef } from '@angular/core';
constructor(private fb: FormBuilder, private _changeDetectionRef: ChangeDetectorRef) {
}
ngAfterViewChecked(): void {
this._changeDetectionRef.detectChanges();
}

The above code worked without issues and errors in console
