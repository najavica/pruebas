I have made a few changes that helped me implement.

DATATABLE-COLUMN-FILTER: assets/js/king-table.js

The example used defines the number of columns in the table to 5. I have updated to add filtering on any number of columns in the table:

/* column filters */
if( $('.datatable-column-filter').length > 0 ) {

var dtTable = $('.datatable-column-filter').DataTable({ // use DataTable, not dataTable
sDom: // redefine sDom without lengthChange and default search box
"t"+
"<'row'<'col-sm-6'i><'col-sm-6'p>>"
});

var cols = dtTable.columns().header().length;

var tr = document.createElement("tr");
tr.className = "row-filter";

for (i = 0; i < cols; i++) {

var th = document.createElement("th");

var inpt = document.createElement("input");
inpt.type = "text";
inpt.className = "form-control input-sm";

th.appendChild(inpt);

tr.appendChild(th);

};

$('.datatable-column-filter thead').append(tr);

MAIN NAVIGATION: assets/js/king-common.js

I replaced $('.main-menu .js-sub-menu-toggle').on('click', function(e){
with $(document).on('click', '.main-menu .js-sub-menu-toggle', function(e){

I have updated this because I add elements to the main-menu via DOM on document.ready, the supplied example doesn't work because the side-bar has already been rendered and when you click on the menu items nothing happens, this fixes that.

$(document).on('click', '.main-menu .js-sub-menu-toggle', function(e){

e.preventDefault();

$li = $(this).parent('li');
if( !$li.hasClass('active')){
$li.find(' > a .toggle-icon').removeClass('fa-angle-left').addClass('fa-angle-down');
$li.addClass('active');
$li.find('ul.sub-menu')
.slideDown(300);
}
else {
$li.find(' > a .toggle-icon').removeClass('fa-angle-down').addClass('fa-angle-left');
$li.removeClass('active');
$li.find('ul.sub-menu')
.slideUp(300);
}
});
