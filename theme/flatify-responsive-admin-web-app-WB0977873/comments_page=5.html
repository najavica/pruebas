So here is what I have found:
Inside <div data-ui-wizard-form> you need to include an additional wrapper <div class="steps-wrapper">

So it should look like this
```
<div data-ui-wizard-form>
<div class="steps-wrapper">
<h1>Initial Set Up</h1>
<div></div>
<h1>Review</h1>
<div></div>
</div>
</div>
```

Then next thing is to add to the "uiWizardForm" directive and attach the form wizard to the controller scope. That way you can run your methods from the controller.
```
.directive("uiWizardForm", function($compile)
{
return {
restrict: 'A',
link: function(scope, ele)
{
steps = ele.children('.steps-wrapper').steps({
onStepChanging: function (event, currentIndex, newIndex)
{
return scope.onStepChanging();
},
onFinishing: function (event, currentIndex)
{
return scope.onFinishing();
},
onFinished: function (event, currentIndex)
{
scope.finishedWizard();
},
});
$compile(steps)(scope);
}
}
});
```
Now everything is working except I am unable to access the parent controller from the ui-wizard-form.
I am unable to access variables or create a ng-repeat with variables created in the parent controller.
Any help would be greatly appreciated.

coltforty PURCHASED
2 years ago
Ok, I got it working for me. I need to create a Reference to scope within the directive. Then assign it in the controller method of the directive.
```
.directive("uiWizardForm", function()
{
var scope;

return {
restrict: "A",
controller:function($scope,$element){
scope = $scope;
},
compile: function($element){
var steps = $element.children('.steps-wrapper').steps({
onStepChanging: function (event, currentIndex, newIndex)
{
return scope.onStepChanging();
},
onFinishing: function (event, currentIndex)
{
return scope.onFinishing();
},
onFinished: function (event, currentIndex)
{
scope.finishedWizard();
}
});
}
};
});
```
