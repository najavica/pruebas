web-master72 SELLER

Hello,

If you already add "current" class to one of the filters then now you need change JS function in "assets/js/custom.js" file.

Find line - "worksgrid.imagesLoaded(function() {"

And replace this function with next code:

var selector = $('.current', filters).attr('data-filter');
worksgrid.imagesLoaded(function() {
worksgrid.isotope({
layoutMode: 'packery',
itemSelector: '.work-item',
transitionDuration: '0.3s',
filter: selector,
packery: {
columnWidth: '.grid-sizer',
},
});
});
